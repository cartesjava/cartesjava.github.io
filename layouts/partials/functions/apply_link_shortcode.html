{{- define "apply_link_shortcode" -}}{{ $params := newScratch }}{{ $args := .Get "args" }}{{ $src := "" }}{{ $text := "" }}{{ if reflect.IsSlice $args }}{{ $src = index $args 0 }}{{ $text = index $args 1 }}{{ else }}{{ $src = $args.Get "src" }}{{ $text = $args.Get "text" }}{{ if not $src }}{{ $src = $args.Get 0 }}{{ end }}{{ if not $text }}{{ $text = $args.Get 1 }}{{ end }}{{ end }}{{ $params.Set "src" $src }}{{ $params.Set "text" $text }}{{ template "link" $params }}{{- end -}}
